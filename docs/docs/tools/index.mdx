import { useState } from 'react';
import { motion } from 'framer-motion';
import { Search } from 'lucide-react';
import styles from "./index.module.scss";
import tools from '@site/src/components/ToolsData';
import React, { useEffect, useRef } from "react";
import ToolCard from '@site/src/components/ToolCard';

export function ToolList() {
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedTags, setSelectedTags] = useState([]);
  const [sortOption, setSortOption] = useState('name');

  const allTags = Array.from(new Set(tools.flatMap((tool) => tool.tags)));
  const filteredTools = tools
    .filter(
      (tool) =>
        tool.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
        tool.description.toLowerCase().includes(searchTerm.toLowerCase())
    )
    .filter(
      (tool) =>
        selectedTags.length === 0 || selectedTags.every((tag) => tool.tags.includes(tag))
    )
    .sort((a, b) => {
      if (sortOption === 'name') {
        return a.name.localeCompare(b.name);
      } else if (sortOption === 'tags') {
        return a.tags.length - b.tags.length;
      }
      return 0;
    });

  const toggleTag = (tag) => {
    if (selectedTags.includes(tag)) {
      setSelectedTags(selectedTags.filter((t) => t !== tag));
    } else {
      setSelectedTags([...selectedTags, tag]);
    }
  };

  return (
    <>
      <div className="mb-6 flex flex-col md:flex-row md:items-center md:justify-between">
        <div className="relative w-full md:w-1/2 mb-4 md:mb-0">
          <input
            type="text"
            placeholder="Search tools..."
            value={searchTerm}
            onChange={(e) => setSearchTerm(e.target.value)}
            className="w-full pl-10 pr-4 py-2 border rounded focus:outline-none focus:ring-2 focus:ring-blue-500"
          />
          <Search className="absolute left-3 top-2.5 text-gray-500" />
        </div>
        <select
          value={sortOption}
          onChange={(e) => setSortOption(e.target.value)}
          className="border rounded p-2"
        >
          <option value="name">Sort by Name</option>
          <option value="tags">Sort by Tags</option>
        </select>
      </div>
      <div className="mb-6 flex flex-wrap gap-2">
        {allTags.map((tag) => {
          const isSelected = selectedTags.includes(tag);
          const tagClass =
            'cursor-pointer px-3 py-1 rounded-full text-sm transition-colors ' +
            (isSelected
              ? 'bg-blue-500 text-white'
              : 'bg-gray-200 text-gray-800 hover:bg-blue-100');
          return (
            <span key={tag} onClick={() => toggleTag(tag)} className={tagClass}>
              {tag}
            </span>
          );
        })}
      </div>
      <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">
        {filteredTools.map((tool) => (
          <ToolCard key={tool.name} tool={tool} />
        ))}
      </div>
    </>
  );
}
